# Remove Columns with >20% Missing Data
threshold = len(df) * 0.2
df_cleaned = df.loc[:, df.isnull().sum() <= threshold]
dropped_columns = df.columns.difference(df_cleaned.columns)
print(f"Columns dropped due to >20% missing data: {list(dropped_columns)}")

# Create a binary missing indicator for each column
missing_indicators = df_cleaned.isnull().astype(int)

# Display missing data indicators
print("Missing Data Indicators:\n", missing_indicators.sum())

# Impute Missing Values (Simple Imputation Strategy)
num_cols = df_cleaned.select_dtypes(include=np.number).columns
cat_cols = df_cleaned.select_dtypes(exclude=np.number).columns

# Mean Imputation for Numerical Columns
num_imputer = SimpleImputer(strategy='mean')
df_cleaned[num_cols] = num_imputer.fit_transform(df_cleaned[num_cols])

# Most Frequent Imputation for Categorical Columns
cat_imputer = SimpleImputer(strategy='most_frequent')
df_cleaned[cat_cols] = cat_imputer.fit_transform(df_cleaned[cat_cols])

# Save the Cleaned Data
df_cleaned.to_csv('preprocessed_kidney_disease.csv', index=False)
print("Preprocessed dataset saved successfully.")

Eventhough some of the attributes were removed there are some more missing values distributed through out the dataset.For the missing numerical values they will be replaced by the mean values of the respective columns.For the missing values ion the catergorical columns the most occured value was replaced for thier respective column.¶

Feature selection is the process which helps to identify the main variable/variables which contribute to the machine learning model.In this research paper Ant Colony Optimization (ACO) has been used.¶

Ant colony optimozation is an optimization algorithm which is built based on the behaviour of ants.Ants choose the shortest path between thier food source and colony by laying pheromones(a chemical substance layed by ants), when pheromones are layed it tends to attract other ants which results in everyone using the same paath.Over time when the same path is used for a particular time period the paths automatically convert into good algorithms where ants find solutions for thier hardships.¶

Similar applications of ACO is Feature selection,Network routing,Traveling sales problem and vehicle routing problem¶

ACO is a very flexible optimization method whilst it can also be a very computationally expensive algorithm¶

import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.ensemble import RandomForestClassifier
from sklearn.tree import DecisionTreeClassifier
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import classification_report, accuracy_score

# For the ACO implementation
class AntColonyOptimizer:
    def __init__(self, n_ants, n_iterations, decay, alpha=1, beta=1):
        self.n_ants = n_ants
        self.n_iterations = n_iterations
        self.decay = decay
        self.alpha = alpha
        self.beta = beta

    def fit(self, X, y):
        n_features = X.shape[1]
        pheromone = np.ones(n_features)
        best_score = -np.inf
        best_solution = None

        for i in range(self.n_iterations):
            solutions = self.generate_solutions(pheromone, n_features)
            scores = self.evaluate_solutions(solutions, X, y)
            best_iteration_score = max(scores)
            best_iteration_solution = solutions[np.argmax(scores)]

            if best_iteration_score > best_score:
                best_score = best_iteration_score
                best_solution = best_iteration_solution

            pheromone = self.update_pheromone(pheromone, solutions, scores)

        return best_solution, best_score

    def generate_solutions(self, pheromone, n_features):
        solutions = []
        for _ in range(self.n_ants):
            solution = []
            for i in range(n_features):
                if np.random.rand() < pheromone[i]:
                    solution.append(i)
            solutions.append(solution)
        return solutions

    def evaluate_solutions(self, solutions, X, y):
        scores = []
        for solution in solutions:
            if len(solution) == 0:
                scores.append(0)
                continue
            X_subset = X[:, solution]
            score = cross_val_score(RandomForestClassifier(), X_subset, y, cv=3).mean()
            scores.append(score)
        return scores

    def update_pheromone(self, pheromone, solutions, scores):
        for i in range(len(pheromone)):
            pheromone[i] *= self.decay
            for j in range(len(solutions)):
                if i in solutions[j]:
                    pheromone[i] += scores[j]
        return pheromone

The main objective of ACO is to increase the accuracy of the model.Overtime the path becomes more optimal.Here is a break ddown of the above code.¶

1.Initialization¶

Initilizes the process by giving pheromone levels for each data¶

2.Iteration process¶

After deploying a popualtion of ants every single ant has a duty to construct a feature subset based on the pheromone level¶

3.Solution generation¶

The more the pheromone level the more it has a chance to be included in the subset.¶

4.Solution Evaluation¶

Once a subset of features are selected a random forest classifier is used to evaluate the model.After the evaluation cross validation is used to check the performance of the model then its accuracy is noted as the fitness of the model.¶

5.Pheromone update¶

After the evaluation the pheromones are again updated¶

6.convergence¶

Next the final solution is been chosed.The highest recorded subset performance is selected.¶
